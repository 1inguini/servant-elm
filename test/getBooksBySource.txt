getBooksBy : Int -> Task.Task Http.Error (Book)
getBooksBy id =
  let request =
        { verb = "GET"
        , headers = [("Content-Type", "application/json")]
        , url = "/books/" ++ (id |> toString |> Http.uriEncode) ++ ""
        , body = Http.empty
        }
  in  Http.fromJson
        decodeBook
        (Http.send Http.defaultSettings request)
