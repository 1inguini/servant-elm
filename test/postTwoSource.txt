postTwo : String -> Task.Task Http.Error (Maybe Int)
postTwo body =
  let request =
        { verb = "POST"
        , headers = [("Content-Type", "application/json")]
        , url = "/two"
        , body = Http.string (JS.encode 0 (JS.string body))
        }
  in  Http.fromJson
        (maybe int)
        (Http.send Http.defaultSettings request)
